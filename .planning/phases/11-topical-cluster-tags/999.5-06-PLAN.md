**Step 2 — Create a small inline helper** (or inline JSX) for the dots prefix. Prefer a helper function INSIDE the component to avoid inline duplication across 3 sites:

```typescript
function renderClusterDots(tags: string[] | undefined) {
  if (!tags || tags.length === 0) return null;
  const visible = tags.slice(0, 3);
  return (
    <span className="inline-flex items-center gap-0.5 mr-1 flex-shrink-0" data-testid="cluster-dots">
      {visible.map((tag) => {
        const color = getClusterColor(tag);
        return (
          <span
            key={tag}
            className={`inline-block w-1.5 h-1.5 rounded-full ${color.dot}`}
            aria-hidden
          />
        );
      })}
    </span>
  );
}
```

Place it at module top (outside the component, under the imports and helpers block) — pure function, no hooks needed.

