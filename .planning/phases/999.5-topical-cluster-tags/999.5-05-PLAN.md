<objective>
Color same-cluster edges with a muted cluster tone in `LinkCountEdge.tsx`. Read source and target node tags via `useReactFlow().getNode(id)`. When there's ≥1 shared tag, compute the edge color by picking the first shared tag in alphabetical-sorted order, then call `getClusterColor(tag).edge` to get the hex. Cross-cluster or unassigned edges retain the current default gray. Selected edges keep overriding to indigo.

Purpose: Visual confirmation that the PageRank cluster bonus is actually firing for the expected edges. Reinforces topical clusters as a first-class visual primitive.

Output: Canvas edges between same-cluster nodes render in their cluster's muted color. New dedicated edge-level test file asserts the stroke color decision.
</objective>

- `LinkCountEdge` reads source and target node tags at render time via `useReactFlow().getNode(source).data.tags` and same for target
- Deterministic color: sort shared tags alphabetically, pick the FIRST one
- Default gray when no shared tag or either node missing
- Selected indigo still wins over cluster color
- Memoize the color computation so React Flow re-renders don't re-hash on every frame
- New test file `LinkCountEdge.test.tsx` mocks `useReactFlow` to return tagged source+target nodes and asserts `BaseEdge` `style.stroke` matches the expected cluster hex (MAJOR-2 fix)

**Step 3 — Compute cluster color inside the component** (after the existing `getSmoothStepPath` call at ~27-34):
```typescript
const { getNode } = useReactFlow();

const clusterStrokeHex = useMemo(() => {
  const sourceNode = getNode(source) as import('reactflow').Node<UrlNodeData> | undefined;
  const targetNode = getNode(target) as import('reactflow').Node<UrlNodeData> | undefined;
  const sourceTags = sourceNode?.data.tags ?? [];
  const targetTags = targetNode?.data.tags ?? [];

  // Find shared tags
  const sourceSet = new Set(sourceTags);
  const shared = targetTags.filter((t) => sourceSet.has(t));
  if (shared.length === 0) return null;

  // Deterministic rule: first shared tag in alphabetical-sorted order
  shared.sort();
  return getClusterColor(shared[0]).edge;
}, [source, target, getNode]);
```

NOTE on dependencies: `getNode` from `useReactFlow()` is a stable reference across renders in React Flow v11+, so it's safe in the dep array. If stale data issues occur, add a dummy nonce from a context or switch to a per-edge data prop (fallback — requires App.tsx wiring change, AVOID).

**Step 5 — Create the dedicated test file `packages/web/src/components/LinkCountEdge.test.tsx`** (MAJOR-2 fix). Recommended structure:

```typescript
import { describe, it, expect, vi, beforeEach } from 'vitest';
import { render } from '@testing-library/react';
import type { Node } from 'reactflow';
import { Position } from 'reactflow';
import type { UrlNodeData } from '../lib/graph-utils';
import { getClusterColor } from '../lib/cluster-colors';

// Mock useReactFlow so we control getNode
const nodeMap = new Map<string, Node<UrlNodeData>>();
vi.mock('reactflow', async () => {
  const actual = await vi.importActual<typeof import('reactflow')>('reactflow');
  return {
    ...actual,
    useReactFlow: () => ({
      getNode: (id: string) => nodeMap.get(id),
    }),
  };
});

import { LinkCountEdge } from './LinkCountEdge';

function makeNode(id: string, tags: string[] | undefined): Node<UrlNodeData> {
  return {
    id,
    type: 'urlNode',
    position: { x: 0, y: 0 },
    data: { urlTemplate: `/${id}`, pageCount: 1, ...(tags && { tags }) },
  } as Node<UrlNodeData>;
}

function renderEdge(props: { source: string; target: string; selected?: boolean }) {
  const { source, target, selected = false } = props;
  return render(
    <svg>
      <LinkCountEdge
        id={`${source}-${target}`}
        source={source}
        target={target}
        sourceX={0}
        sourceY={0}
        targetX={100}
        targetY={100}
        sourcePosition={Position.Bottom}
        targetPosition={Position.Top}
        data={{ linkCount: 1 } as never}
        selected={selected}
        style={{}}
        markerEnd={undefined}
      />
    </svg>,
  );
}

describe('LinkCountEdge stroke color', () => {
  beforeEach(() => {
    nodeMap.clear();
  });

  it('uses default gray when neither node has tags', () => {
    nodeMap.set('a', makeNode('a', undefined));
    nodeMap.set('b', makeNode('b', undefined));
    const { container } = renderEdge({ source: 'a', target: 'b' });
    const path = container.querySelector('path[stroke]') as SVGPathElement | null;
    expect(path?.getAttribute('stroke')).toBe('#9CA3AF');
  });

  it('uses cluster color when source and target share a tag', () => {
    nodeMap.set('a', makeNode('a', ['food']));
    nodeMap.set('b', makeNode('b', ['food']));
    const { container } = renderEdge({ source: 'a', target: 'b' });
    const path = container.querySelector('path[stroke]') as SVGPathElement | null;
    expect(path?.getAttribute('stroke')).toBe(getClusterColor('food').edge);
  });

  it('picks first shared tag alphabetically when multiple overlap', () => {
    nodeMap.set('a', makeNode('a', ['taipei', 'food']));
    nodeMap.set('b', makeNode('b', ['food', 'taipei']));
    const { container } = renderEdge({ source: 'a', target: 'b' });
    const path = container.querySelector('path[stroke]') as SVGPathElement | null;
    expect(path?.getAttribute('stroke')).toBe(getClusterColor('food').edge);
  });

  it('selected edge uses indigo, overriding cluster color', () => {
    nodeMap.set('a', makeNode('a', ['food']));
    nodeMap.set('b', makeNode('b', ['food']));
    const { container } = renderEdge({ source: 'a', target: 'b', selected: true });
    const path = container.querySelector('path[stroke]') as SVGPathElement | null;
    expect(path?.getAttribute('stroke')).toBe('#6366F1');
  });

  it('falls back to gray when a node is missing', () => {
    nodeMap.set('a', makeNode('a', ['food']));
    // target 'b' not set
    const { container } = renderEdge({ source: 'a', target: 'b' });
    const path = container.querySelector('path[stroke]') as SVGPathElement | null;
    expect(path?.getAttribute('stroke')).toBe('#9CA3AF');
  });
});
```
