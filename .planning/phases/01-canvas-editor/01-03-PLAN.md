  <behavior>
**1. createDefaultNode(position: { x: number; y: number }): Node&lt;UrlNodeData&gt;**
- Input: `{ x: 100, y: 200 }` -> Output: `{ id: "node-1", type: "urlNode", position: { x: 100, y: 200 }, data: { urlTemplate: "/page/<id>", pageCount: 1 } }`
- Each call produces a unique incremented id: "node-1", "node-2", "node-3"...
- Default urlTemplate is `/page/<id>` (per D-03)
- Default pageCount is `1` (per D-03)
- type is always `"urlNode"`

**2. updateNodeData(nodes: Node[], nodeId: string, newData: Partial&lt;UrlNodeData&gt;): Node[]**
- Input: `[nodeA, nodeB]`, `"a"`, `{ pageCount: 50 }` -> Output: new array where nodeA.data.pageCount=50, nodeB unchanged
- Returns a NEW array (immutable)
- Only the matching node's data is merged; all other nodes remain identical (referential equality)
- If nodeId not found, returns array unchanged

**3. updateEdgeLinkCount(edges: Edge[], edgeId: string, linkCount: number): Edge[]**
- Input: `[edgeX, edgeY]`, `"x"`, `5` -> Output: new array where edgeX.data.linkCount=5, edgeY unchanged
- Returns a NEW array (immutable)
- If edgeId not found, returns array unchanged

**4. validateNodeData(data: { urlTemplate: string; pageCount: number }): string | null**
- `{ urlTemplate: "", pageCount: 1 }` -> `"URL template cannot be empty"` (matches UI-SPEC copy exactly)
- `{ urlTemplate: "  ", pageCount: 1 }` -> `"URL template cannot be empty"` (whitespace-only = empty)
- `{ urlTemplate: "/page/<id>", pageCount: 0 }` -> `"Page count must be at least 1"`
- `{ urlTemplate: "/page/<id>", pageCount: -5 }` -> `"Page count must be at least 1"`
- `{ urlTemplate: "/page/<id>", pageCount: 1 }` -> `null` (valid)
- `{ urlTemplate: "/page/<id>", pageCount: 100 }` -> `null` (valid)

**5. validateLinkCount(count: number): number**
- `5` -> `5` (pass-through valid)
- `0` -> `1` (clamp to min)
- `-3` -> `1` (clamp to min)
- `NaN` -> `1` (fallback to default)
- `1.7` -> `1` (floor to integer -- link counts are whole numbers)

**6. formatPageCount(n: number): string**
- `1` -> `"1 page"` (singular)
- `0` -> `"0 pages"` (plural for zero)
- `100` -> `"100 pages"` (plural)
- `2` -> `"2 pages"` (plural)

```typescript
import type { Node, Edge } from 'reactflow';

export interface UrlNodeData {
  urlTemplate: string;
  pageCount: number;
}

export interface LinkCountEdgeData {
  linkCount: number;
}

let nodeIdCounter = 0;

export function resetNodeIdCounter(): void {
  nodeIdCounter = 0;
}

export function createDefaultNode(position: { x: number; y: number }): Node<UrlNodeData> {
  nodeIdCounter += 1;
  return {
    id: `node-${nodeIdCounter}`,
    type: 'urlNode',
    position,
    data: {
      urlTemplate: '/page/<id>',
      pageCount: 1,
    },
  };
}

export function updateNodeData(
  nodes: Node<UrlNodeData>[],
  nodeId: string,
  newData: Partial<UrlNodeData>,
): Node<UrlNodeData>[] {
  return nodes.map((n) =>
    n.id === nodeId ? { ...n, data: { ...n.data, ...newData } } : n,
  );
}

export function updateEdgeLinkCount(
  edges: Edge<LinkCountEdgeData>[],
  edgeId: string,
  linkCount: number,
): Edge<LinkCountEdgeData>[] {
  return edges.map((e) =>
    e.id === edgeId ? { ...e, data: { ...e.data, linkCount } } : e,
  );
}

export function validateNodeData(data: {
  urlTemplate: string;
  pageCount: number;
}): string | null {
  if (data.urlTemplate.trim() === '') {
    return 'URL template cannot be empty';
  }
  if (data.pageCount < 1) {
    return 'Page count must be at least 1';
  }
  return null;
}

export function validateLinkCount(count: number): number {
  if (Number.isNaN(count) || count < 1) {
    return 1;
  }
  return Math.floor(count);
}

export function formatPageCount(n: number): string {
  return n === 1 ? '1 page' : `${n} pages`;
}
```
